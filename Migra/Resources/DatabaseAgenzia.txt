**AVVIO PROGRAMMA DI IMPORTAZIONE ...;

USE MASTER;

IF EXISTS (SELECT name FROM sys.databases WHERE name = N'DB{agenzia}')
   DROP DATABASE DB{agenzia};

**creazione del database ...;

CREATE DATABASE DB{agenzia}  
ON   
(   NAME = DB{agenzia},  
    FILENAME = '{0}\DB{agenzia}.mdf',  
    SIZE = 100MB,  
    MAXSIZE = 4GB,  
    FILEGROWTH = 10MB )  
LOG ON  
(   NAME = LG{agenzia},  
    FILENAME = '{0}\LG{agenzia}.log',  
    SIZE = 50MB,  
    MAXSIZE = 1GB,  
    FILEGROWTH = 25MB );

USE DB{agenzia};
ALTER DATABASE [DB{agenzia}] SET AUTO_CLOSE OFF;
